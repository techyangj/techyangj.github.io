Fabric搭建

安装python2.7

安装git vim curl build-essential

sudo apt install git
sudo apt install vim
sudo apt install curl
sudo apt-get install build-essential

安装nodejs

wget https://nodejs.org/dist/v14.17.6/node-v14.17.6-linux-x64.tar.xz
sudo tar -xvf node-v14.17.6-linux-x64.tar.xz -C /opt
sudo vim /etc/profile
export NODEJS_HOME=/opt/node-v14.17.6-linux-x64
export PATH=$PATH:$NODEJS_HOME/bin
. /etc/profile

node -v

安装go

mkdir $HOME/go
配置变量：
vim ~/.bashrc
export GOROOT=/usr/local/go
export GOPATH=$HOME/go
export PATH=$PATH:$GOROOT/bin:$GOPATH/bin
重新加载：
source ~/.bashrc

go version

go env -w GO111MODULE=on
go env -w GOPROXY=https://goproxy.cn,direct

fabric源码安装

mkdir -p $GOPATH/src/github.com/hyperledger/
cd $GOPATH/src/github.com/hyperledger
git clone https://github.com/hyperledger/fabric.git
git clone https://github.com/hyperledger/fabric-ca.git
#进入到fabric文件下,下载fabric-samples镜像
cd fabric
git checkout -b v2.2.0 #切换到2.2.0,,默认为master分支
git branch #查看当前分支
git clone https://github.com/hyperledger/fabric-samples.git
cd fabric-samples
git checkout -b v2.2.0
git branch
